Algorithm 1: State space search transition model

Input: A node, node, that corresponds to a state

Output: A list of successor nodes to node


    procedure expand(node)
        successors ← ∅
        if node.microState then
            return expandMicroState(node, successors)
        else
            return expandMacroState(node, successors)
    end procedure

    procedure expandMicroState(node, successors)
        make new nodes, node1, node2, node3,  from node
         node1.turnRight
         node2.turnLeft
         node1.pathCost ← node.pathCost + 1
         node2.pathCost ← node.pathCost + 1
         successors ← successors ∪ {node1, node2}
        if node.goForward  is a valid action then
            node3.goForward
            node3.pathCost ← node.pathCost + 1
            successors ← successors ∪ {node3}
        return successors
    end procedure

    procedure expandMacroState(node, successors)
        make node1, node2, node3, node4 from node
        if node.goLeft is a valid action then
            node1.goLeft
            node1.pathCost ← node.pathCost + 1
            successors ← successors ∪ {node1}
        if node.goRight is a valid action then
            node2.goRight
            node2.pathCost ← node.pathCost + 1
            successors ← successors ∪ {node2}
        if node.goTop is a valid action then
            node3.goTop
            node3.pathCost ← node.pathCost + 1
            successors ← successors ∪ {node3}
        if node.goBottom is a valid action then
            node4.goBottom
            node4.pathCost ← node.pathCost + 1
            successors ← successors ∪ {node4}
        return successors
    end procedure


Algorithm 2: Local search transition model without heuristic

Input: An agent a and current state s

Output: A successor state s′


    procedure successorState(a,s)
        successors ← ∅
        foreach state s′ ∈ succ(s) do
            if s′ has not been visited then
                successors ← successors ∪ {s′}
        if successors s is empty then
            randomly select an s′ from succ(s)
        else
            randomly select an s′ from successors
        return s′
    end procedure
